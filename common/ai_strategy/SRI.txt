SRI_befriend_GER = {
	allowed = {
		tag = SRI
	}
	enable = {
		has_completed_focus = SRI_international_proximity
		SRI = {
			has_government = socialism
		}
		GER = {
			OR = {
				has_government = communism
				has_government = socialism
			}
			has_completed_focus = GER_right
		}
	}
	abort = {
		OR = {
			NOT = {
				country_exists = GER
			}
			has_war_with = GER
			GER = {
				has_completed_focus = GER_left
			}
			is_subject = yes
		}
	}
	ai_strategy = {
		type = diplo_action_desire
		id = GER
		target = improve_relation
		value = 200
	}
}
