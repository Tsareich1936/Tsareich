        target_trigger = {
            FROM = {
                oil > 30
                is_owned_by = RUS
            }
        }
        available = {

        }
        visible = {
            has_country_flag = RUS_rubber_factory_flag
        }
        icon = GFX_decision_rubber

		cost = 100
        fire_only_once = yes
        ai_will_do = { factor = 2000 }
        days_remove = 90
        complete_effect = {
            set_temp_variable = { temp1 = FROM.resource@oil }

                set_state_flag = RUS_state_rubber_factory_build
            }
        }
    }
