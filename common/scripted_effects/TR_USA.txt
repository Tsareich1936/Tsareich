		limit = {

				tag = TXS
				tag = DST
				tag = IRC
			}
		}
		add_to_array = {
			array = global.second_american_civil_war
		}
	}
	every_country = {
		limit = {
			is_in_array = {
				global.second_american_civil_war = THIS
			}
		}
		PREV = {
			declare_war_on = {
				target = PREV
				type = civil_war
			}
			if = {
				limit = {
					PREV = {
						NOT = {
							OR = {
								tag = TXS
								tag = IRC
							}
						}
					}
					THIS = {
						NOT = {
							OR = {
								tag = TXS
								tag = IRC
							}
						}
					}
				}
				add_civil_war_target = PREV
			}
			PREV = {
				add_to_war = {
					targeted_alliance = THIS
					enemy = PREV
				}
			}
		}
